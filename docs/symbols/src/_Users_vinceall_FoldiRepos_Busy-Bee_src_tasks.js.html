1 function handleMessage(e) {
  2 
  3   var data = e.data;
  4 
  5   if (!data.cmd) {
  6     return;
  7   }
  8 
  9   switch (data.cmd) {
 10     case 'add':
 11       add(data.workerId, data.start, data.options);
 12       break;
 13 
 14     case 'sub':
 15       sub(data.workerId, data.start, data.options);
 16       break;
 17 
 18     default:
 19       postMessage({cmd: 'error', workerId: data.workerId, start: data.start,
 20           msg: 'Command \"' + data.cmd + '\" does not exist in tasks.js.'});
 21       break;
 22   }
 23 
 24 }
 25 
 26 function add(workerId, start, options) {
 27 
 28   var n = 0, max = options.max;
 29 
 30   while (n < max) {
 31     n++;
 32   }
 33 
 34   postMessage({cmd: 'done', workerId: workerId, start: start});
 35 }
 36 
 37 function sub(workerId, start, options) {
 38 
 39   var n = options.max;
 40 
 41   while (n > 0) {
 42     n--;
 43   }
 44 
 45   postMessage({cmd: 'done', workerId: workerId, start: start});
 46 }
 47 
 48 self.addEventListener('message', handleMessage, false);
