1 /*global window, exports */
  2 /**
  3  * Creates a new Project.
  4  *
  5  * A BusyBee project uses a pool of Web Workers to complete a list
  6  * of tasks. Tasks are passed into a new Project as an array of objects
  7  * with a command and a set of options. Supply callbacks for feedback on start,
  8  * progress, completion and error. All callbacks are triggered via the event object
  9  * from the Web Worker.
 10  *
 11  * @example
 12  * var tasks = [
 13  *    {
 14  *      cmd: 'add', options: {max: 20000}
 15  *    },
 16  *    {
 17  *      cmd: 'sub', options: {max: 20000}
 18  *    }
 19  * ]
 20  * var project = new BusyBee.Project(
 21  *    tasks,
 22  *    function (e) {
 23  *      console.log('Work is starting.');
 24  *    },
 25  *    function (e) {
 26  *      console.log('Work is progressing.');
 27  *    },
 28  *    function (e) {
 29  *      console.log('Work is finished!');
 30  *    },
 31  *    function (e) {
 32  *      console.log('We encountered an error: ' + e.data.msg);
 33  *    }
 34  * );
 35  * project.start();
 36  *
 37  * @constructor
 38  * @param {Array.<Object>} tasks An array of tasks to send to the Web Workers.
 39  * @param {Function} opt_onStart A function to call when starting Web Workers.
 40  * @param {Function} opt_onProgress A function to call after each Web Worker completes.
 41  * @param {Function} opt_onFinish A function to call after completing all tasks.
 42  * @param {Function} opt_onError A function to call if Workers encounter an error.
 43  * @param {string} [opt_path = 'scripts/tasks.js'] The path to the tasks.js Worker file.
 44  * @param {Swarm} opt_swarm An instance of the Swarm class. Supply an instance if you
 45  *    want to alter number of Workers from the default value of 4.
 46  */
 47 function Project(tasks, opt_onStart, opt_onProgress, opt_onFinish, opt_onError, opt_path, opt_swarm) {
 48 
 49   this.tasks = tasks;
 50   this.onStart = opt_onStart || this._noop;
 51   this.onProgress = opt_onProgress || this._noop;
 52   this.onFinish = opt_onFinish || this._noop;
 53   this.onError = opt_onError || this._noop;
 54   this.path = opt_path || 'scripts/tasks.js';
 55   this.swarm = opt_swarm || new exports.Swarm(4);
 56 
 57   this.startTime = new Date().getTime();
 58 
 59   this.swarm._init(this.tasks, this.path, this.onStart.bind(this), this.onProgress.bind(this),
 60       this.onFinish.bind(this), this.onError.bind(this));
 61 }
 62 
 63 /**
 64  * Call to detect if this browser supports Web Workers.
 65  *
 66  * @returns {boolean} True if browser supports Web Workers. False if not.
 67  * @public
 68  */
 69 Project.workersAvailable = function() {
 70   return !!window.Worker;
 71 };
 72 
 73 Project.prototype.name = 'Project';
 74 
 75 /**
 76  * Call to start the project.
 77  *
 78  * @public
 79  */
 80 Project.prototype.start = function() {
 81   this.swarm._start();
 82 };
 83 
 84 /**
 85  * An empty function to use as a default callback.
 86  */
 87 Project.prototype._noop = function() {};
 88 
 89 exports.Project = Project;
